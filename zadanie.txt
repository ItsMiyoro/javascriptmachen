const product ={
    name: "Laptop Dell Vostro 15 3510 NVMe DDR4 I3 NobaraOS",
    price: 1379.99,
    stock: 2,
    category: "Laptop",
    specifications :{
        ram: "24gb (8+16) 3200mhz",
        disk: "1tb Lexar 620nm 3300 / 3000",
        cpu: "Intel Core I3-115G4"
    },
    displayProduct: function(){
        console.log(product.name+" za "+product.price+"zł. Kategorii produktu : "+product.category+". Stan w magazynie : "+product.stock);
    },
    isAvaible: function(){
        if (product.stock>0){
            console.log(true);
        }
        else{
            console.log(false);
        }
    },
    sell: function(){
        if (product.stock>0){
            console.log("Sprzedano produkt "+product.name);
            console.log("Aktualny stan : "+product.stock);
            product.stock = product.stock-1;
        }
        else{
            console.log("Już nie ma produktu"+product.name+" w magazynie")
        }
    }
}
product.displayProduct();
product.isAvaible();
product.sell();
